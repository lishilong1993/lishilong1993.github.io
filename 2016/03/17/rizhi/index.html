JAVA常用算法手册之阅读笔记<一>

  • 笔记

数据结构：

数据：信息的载体；整数、字符、字符串、图像、声音

数据元素：数据的基本单位，一个数据元素由若干个数据项组成；

数据项：具有独立含义的最小标识单位，可称为：字段、域、属性

数据结构：数据之间的相互关系，数据的组织形式

内容：

1、数据的逻辑结构

2、数据的存储结构

3、数据的运算

数据结构分类：

1、线性结构

2、非线性结构

存储方式：

1、顺序存储

2、链接存储

3、索引存储

4、散列存储

数据类型：

1、基本数据类型

2、聚合数据类型

ADT:抽象数据类型

数据抽象、数据封装

常用数据结构：

1、数组（Array）

2、栈（Stack）:特殊的线性表；按照后进先出的原则存储数据；没有数据的栈为空栈；

3、队列（Queue）:特殊的线性表；按照先进先出的原则存储数据；没有数据的队列为空队列；

4、链表（Linked List）:一种数据元素按照链式存储结构进行存储的数据结构，特点为非连续；链表由系列数据结点构成，每个数据结点包括数据域和引用域；引用域保存数据结构下一个元素的存储地址。

5、树（Tree）:非线性结构；只有一个根节点

6、图（Graph）:非线性结构；数据节点成为顶点，边是顶点的有序偶对；顶点间有边表示两个顶点具有相邻关系。

7、堆（Heap）:树型据结构，一般讨论二叉堆；特点：根节点的值是所有节点中最小或最大的，根节点的两个子树也是堆结构

8、散列表（Hash）:如果在结构中存在关键字和T相等的记录，必定在F(T)的存储位置可以找到该记录。可以不用进行比较直接取得记录。

顺序表操作实例代码

package cn.sunline.oa.test;

import java.util.Scanner;

public class test02List {

public static void main(String[] args) {
    SLType SL=new SLType();

    System.out.println("順序表操作演示：");
    SL.SLInit(SL);
    System.out.println("初始化完成。。");

    Scanner sc=new Scanner(System.in);

    do{
        System.out.println("输入添加的节点信息（学号 姓名 年龄）：");
        DATA data=new DATA();
        data.key=sc.next();
        data.name=sc.next();
        data.age=sc.nextInt();

        if(data.age!=0){
            if(SL.SLAdd(SL, data)==0){
                break;
            }
        }else{
            break;
        }

    }while(true);
    System.out.println("顺序表中的节点顺序为：");
    SL.SLAll(SL);


    System.out.println("要取出节点的序号：");
    int i=sc.nextInt();

    DATA pdata=SL.SLFindByNum(SL, i);
    if(pdata!=null){
        System.out.println("第"+i+"个节点为："+pdata.key+","+pdata.name+","+pdata.age);
    }


    System.out.println("要查找的节点关键字：");
    String key=sc.next();
    int j=SL.SLFindByCont(SL, key);
    DATA pdata1=SL.SLFindByNum(SL, j);
    if(pdata1!=null){
        System.out.println("关键字"+key+"为第"+j+"个节点，该节点为："+pdata1.key+","+pdata1.name+","+pdata1.age);
    }

    System.out.println("插入节点(节点位置，学号，姓名，年龄)：");
    int l=sc.nextInt();
    DATA data2=new DATA();
    data2.key=sc.next();
    data2.name=sc.next();
    data2.age=sc.nextInt();
    SL.SLInsert(SL, l, data2);

    System.out.println("所有节点：");
    SL.SLAll(SL);
}
}

class DATA{ //节点关键字
 String key;
 String name;
 int age;
}

class SLType{ //定义顺序表结构
static final int MAXLEN=100;
DATA[] ListData=new DATA[MAXLEN+1];//保存顺序表的结构数组
int ListLen;            //顺序表已存在节点的数量

void SLInit(SLType SL){//初始化顺序表
    SL.ListLen=0;      //初始化为空表
}

int SLLength(SLType SL){
    return SL.ListLen;   //返回顺序表的元素数量
}

//插入节点
int SLInsert(SLType SL,int n,DATA data){
    int i;

    if(SL.ListLen>=MAXLEN){ //顺序表节点数量已超过最大数量
         System.out.println("顺序表已满，不能插入节点");
         return 0;
    }
    if(n<1||n>SL.ListLen-1){
        System.out.println("插入元素序号错误，不能插入元素！");
        return 0;
    }
    for(i=SL.ListLen;i>=n;i--){ //将顺序表中的数据向后移动
        SL.ListData[i+1]=SL.ListData[i];
    }
    SL.ListData[n]=data; //插入节点
    SL.ListLen++;  //顺序表节点数量增加1
    return 1;
}

//追加节点
int SLAdd(SLType SL,DATA data){ //增加元素到顺序表尾部
    if(SL.ListLen>=MAXLEN){ 
        System.out.println("顺序表已满，不能再添加节点！");
        return 0;
    }
    SL.ListData[++SL.ListLen]=data;
    return 1;
}

//删除节点
int SLDelete(SLType SL,int n){

    if(n<1||n>SL.ListLen+1){ //删除节点的序号不正确
        System.out.println("删除节点序号错误，不能删除节点");
        return 0;
    }
    for(int i=n;i<SL.ListLen;i++){ //将顺序表中的数据向前移动
        SL.ListData[i]=SL.ListData[i+1];
    }
    SL.ListLen--;
    return 1;
}

//根据序号查找节点
DATA SLFindByNum(SLType SL,int n){
    if(n<1||n>SL.ListLen+1){
        System.out.println("节点序号错误，不能返回节点");
        return null;
    }
    return SL.ListData[n];

}

//根据关键字查找节点
int SLFindByCont(SLType SL,String key){
    for(int i=1;i<=SL.ListLen;i++){
        if(SL.ListData[i].key.compareTo(key)==0){
            return i;
        }
    }
    return 0;
}

//显示表中的所有节点
int SLAll(SLType SL){
    for(int i=1;i<=SL.ListLen;i++){
        System.out.println(SL.ListData[i].key+","+SL.ListData[i].name+","+SL.ListData[i].age);
    }
    return 0;
}
}
